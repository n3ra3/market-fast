# ── Репрайсер (продажа: держим мои лоты на 1-м месте) ──────────────────────

REPRICER_ENABLED=1

# Шаг: на сколько units ставить ниже конкурента (1 = минимальный).

# Формат цены в ответах bid-ask: value (десятичная валюта '441.5900') или units.

# ── Keep-alive продаж («онлайн»-статус) ────────────────────────────────────
# Без периодического ping маркет уводит аккаунт в офлайн и снимает лоты с
# продажи. Пингуем POST /ping-new каждые PING_INTERVAL_SEC (< 180с — лимит).
SALE_PING_ENABLED=1
PING_INTERVAL_SEC=150
# /ping-new требует Steam webapi_token. Бот получает его САМ по куке
# steamLoginSecure и обновляет, пока кука жива. Кука — чувствительная Steam-
# сессия: держи её ТОЛЬКО здесь (.env в .gitignore), НЕ коммить.
# Как достать: залогинься на steamcommunity.com -> DevTools -> Application ->
# Cookies -> скопируй ПОЛНОЕ значение cookie "steamLoginSecure".
STEAM_LOGIN_SECURE=your_steamLoginSecure_cookie_here
# Как часто обновлять токен (сек). Токен живёт 24ч — 12ч с запасом.
STEAM_TOKEN_TTL_SEC=43200
# Необязательный прокси для валидации токена Steam'ом. Нужен, если ping-new
# стабильно отдаёт 502 с IP дата-центра (Render): нужен residential-прокси.
# STEAM_PROXY=http://login:pass@ip:port
# Повторы при транзиентном 502/сетевом сбое внутри одного цикла пинга.
PING_RETRIES=3
PING_RETRY_DELAY_SEC=5
